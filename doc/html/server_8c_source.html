735  int s;
736  struct addrinfo hints;

738 
739  memset(&hints, 0, sizeof(struct addrinfo));
740  hints.ai_family = AF_UNSPEC; /* Allow IPv4 or IPv6 */
741  hints.ai_socktype = SOCK_DGRAM; /* Coap uses UDP */
742  hints.ai_flags = AI_PASSIVE | AI_NUMERICHOST;
743 
744  s = getaddrinfo(node, port, &hints, &result);
745  if ( s != 0 ) {
746  fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(s));
747  return NULL;
748  }
749 
750  /* iterate through results until success */
751  for (rp = result; rp != NULL; rp = rp->ai_next) {
752  coap_address_t addr;
753 
754  if (rp->ai_addrlen <= sizeof(addr.addr)) {
755  coap_address_init(&addr);
756  addr.size = rp->ai_addrlen;
757  memcpy(&addr.addr, rp->ai_addr, rp->ai_addrlen);
758 
759  ctx = coap_new_context(&addr);
760  if (ctx) {
761  /* TODO: output address:port for successful binding */
762  goto finish;
763  }
764  }
765  }
766 
767  fprintf(stderr, "no context available for interface '%s'\n", node);
768 
769  finish:
770  freeaddrinfo(result);
771  return ctx;

774 int
775 main(int argc, char **argv) {
776  coap_context_t *ctx;
777  fd_set readfds;
778  struct timeval tv, *timeout;
779  int result;
780  coap_tick_t now;
781  coap_queue_t *nextpdu;
782  char addr_str[NI_MAXHOST] = "::";
783  char port_str[NI_MAXSERV] = "5683";
784  int opt;
785  coap_log_t log_level = LOG_WARNING;
786 
787 #if OEM_DEFINED
788  socket_init();
789 
790  file = fopen("/tmp/log.txt", "w+");
791 
792  pthread_t ptId = 0;
793  if(pthread_mutex_init(&m_lock, NULL) != 0)
794  {
795  printf("%s:pthread_mutex_init error!\n", __func__);
796  }
797 #endif
798 
799  while ((opt = getopt(argc, argv, "A:p:v:")) != -1) {
800  switch (opt) {
801  case 'A' :
802  strncpy(addr_str, optarg, NI_MAXHOST-1);
803  addr_str[NI_MAXHOST - 1] = '\0';
804  break;
805  case 'p' :
806  strncpy(port_str, optarg, NI_MAXSERV-1);
807  port_str[NI_MAXSERV - 1] = '\0';
808  break;
809  case 'v' :
810  log_level = strtol(optarg, NULL, 10);
811  break;
812  default:
813  usage( argv[0], PACKAGE_VERSION );
814  exit( 1 );
815  }
816  }
817 
818  coap_set_log_level(log_level);
819 
820  ctx = get_context(addr_str, port_str);
821  if (!ctx)
822  return -1;
823 
824  init_resources(ctx);
825 
826  signal(SIGINT, handle_sigint);
827 
828  while ( !quit ) {
829  FD_ZERO(&readfds);
830  FD_SET( ctx->sockfd, &readfds );
831 
832  nextpdu = coap_peek_next( ctx );
833 
834  coap_ticks(&now);
835  while (nextpdu && nextpdu->t <= now - ctx->sendqueue_basetime) {
836  coap_retransmit( ctx, coap_pop_next( ctx ) );
837  nextpdu = coap_peek_next( ctx );
838  }
839 
840  if ( nextpdu && nextpdu->t <= COAP_RESOURCE_CHECK_TIME ) {
841  /* set timeout if there is a pdu to send before our automatic timeout occurs */
842  tv.tv_usec = ((nextpdu->t) % COAP_TICKS_PER_SECOND) * 1000000 / COAP_TICKS_PER_SECOND;
843  tv.tv_sec = (nextpdu->t) / COAP_TICKS_PER_SECOND;
844  timeout = &tv;
845  } else {
846  tv.tv_usec = 0;
847  tv.tv_sec = COAP_RESOURCE_CHECK_TIME;
848  timeout = &tv;
849  }
850  result = select( FD_SETSIZE, &readfds, 0, 0, timeout );
851 
852  if ( result < 0 ) { /* error */
853  if (errno != EINTR)
854  perror("select");
855  } else if ( result > 0 ) { /* read from socket */
856  if ( FD_ISSET( ctx->sockfd, &readfds ) ) {
857 #if OEM_THREAD
858  pthread_create(&ptId, NULL, pthread_func, (void *)ctx);
859  usleep(1);
860 #else
861  coap_read( ctx ); /* read received data */
862 #endif //OEM_THREAD
863  /* coap_dispatch( ctx ); /\* and dispatch PDUs from receivequeue *\/ */
864  }
865  } else { /* timeout */
866  if (time_resource) {
867  time_resource->dirty = 1;
868  }
869  }
870 
871 #ifndef WITHOUT_ASYNC
872  /* check if we have to send asynchronous responses */
873  check_async(ctx, ctx->endpoint, now);
874 #endif /* WITHOUT_ASYNC */
875 
876 #ifndef WITHOUT_OBSERVE
877  /* check if we have to send observe notifications */
878  coap_check_notify(ctx);
879 #endif /* WITHOUT_OBSERVE */
880  }
881 
882 #if OEM_DEFINED
883  fclose(file);
884  close(sock);
885 #endif //#if OEM_DEFINED
886  coap_free_context( ctx );
887 
888  return 0;
889 }
