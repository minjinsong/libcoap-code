96 hnd_get_time(coap_context_t *ctx, struct coap_resource_t *resource,
97  coap_address_t *peer, coap_pdu_t *request, str *token,
98  coap_pdu_t *response) {
99  unsigned char buf[2];
100  coap_tick_t now;
101  coap_tick_t t;
102 
103  /* if my_clock_base was deleted, we pretend to have no such resource */
104  response->hdr->code = COAP_RESPONSE_CODE(205),
105 
106  coap_add_option(response, COAP_OPTION_CONTENT_FORMAT,
107  coap_encode_var_bytes(buf, COAP_MEDIATYPE_TEXT_PLAIN), buf);
108 
109  coap_add_option(response, COAP_OPTION_MAXAGE,
110  coap_encode_var_bytes(buf, 0x01), buf);
111 
112  /* calculate current time */
113  coap_ticks(&t);
114  now = my_clock_base + (t / COAP_TICKS_PER_SECOND);
115 
116  /* output ticks */
117  response->length += snprintf((char *)response->data,
118  response->max_size - response->length,
119  "%u", (unsigned int)now);
120 }

172 PROCESS_THREAD(coap_server_process, ev, data)
173 {
174  PROCESS_BEGIN();
175 
176  init_coap();
177  init_resources(coap_context);
178 
179  if (!coap_context) {
180  coap_log(LOG_EMERG, "cannot create context\n");
181  PROCESS_EXIT();
182  }
183 
184  /* etimer_set(&notify_timer, 5 * CLOCK_SECOND); */
185  etimer_set(&dirty_timer, 30 * CLOCK_SECOND);
186 
187  while(1) {
188  PROCESS_YIELD();
189  if(ev == tcpip_event) {
190  coap_read(coap_context); /* read received data */
191  coap_dispatch(coap_context); /* and dispatch PDUs from receivequeue */
192  } else if (ev == PROCESS_EVENT_TIMER && etimer_expired(&dirty_timer)) {
193  time_resource->dirty = 1;
194  etimer_reset(&dirty_timer);
195  }
196  }
197 
198  PROCESS_END();
199 }
